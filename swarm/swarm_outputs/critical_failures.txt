```typescript
// Core corruption detection system
class DatabaseCorruptionHandler {
  async detectCorruption(): Promise<CorruptionType | null> {
    try {
      // PRAGMA integrity_check returns 'ok' or corruption details
      const result = await this.db.get('PRAGMA integrity_check');
      if (result.integrity_check !== 'ok') {
        return CorruptionType.STRUCTURAL;
      }
      
      // Quick table validation
      await this.db.get('SELECT COUNT(*) FROM notes LIMIT 1');
      return null;
    } catch (error) {
      return this.classifyCorruption(error.code);
    }
  }

  private classifyCorruption(errorCode: string): CorruptionType {
    switch (errorCode) {
      case 'SQLITE_CORRUPT': // Error 11
        return CorruptionType.DATA_CORRUPTION;
      case 'SQLITE_NOTADB': // Error 26
        return CorruptionType.INVALID_DATABASE;
      case 'SQLITE_CANTOPEN': // Error 14
        return CorruptionType.ACCESS_FAILURE;
      default:
        return CorruptionType.UNKNOWN;
    }
  }
}
```

```typescript
class WALRecoverySystem {
  async initializeWALMode(): Promise<void> {
    // Enable WAL mode for crash recovery
    await this.db.exec('PRAGMA journal_mode=WAL');
    await this.db.exec('PRAGMA wal_autocheckpoint=1000');
    await this.db.exec('PRAGMA synchronous=NORMAL');
  }

  async recoverFromWAL(): Promise<RecoveryResult> {
    try {
      // Force WAL checkpoint to recover pending transactions
      const result = await this.db.get('PRAGMA wal_checkpoint(TRUNCATE)');
      
      if (result.busy === 0 && result.log === 0) {
        return { success: true, recoveredTransactions: result.checkpointed };
      }
      
      // Manual recovery if checkpoint fails
      return await this.performManualWALRecovery();
    } catch (error) {
      throw new WALRecoveryError(error.message);
    }
  }
}
```

```typescript
class DatabaseBackupSystem {
  async createIncrementalBackup(): Promise<string> {
    const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
    const backupPath = path.join(this.backupDir, `notes_${timestamp}.db`);
    
    try {
      // SQLite backup API for consistent snapshots
      const backup = await this.db.backup(backupPath);
      
      // Verify backup integrity immediately
      const backupDb = new Database(backupPath, { readonly: true });
      const integrityCheck = await backupDb.get('PRAGMA integrity_check');
      backupDb.close();
      
      if (integrityCheck.integrity_check !== 'ok') {
        fs.unlinkSync(backupPath);
        throw new Error('Backup integrity verification failed');
      }
      
      return backupPath;
    } catch (error) {
      throw new BackupError(`Backup creation failed: ${error.message}`);
    }
  }

  async recoverFromBackup(backupPath: string): Promise<void> {
    // Show recovery UI
    this.showRecoveryDialog('Recovering from backup...');
    
    try {
      // Validate backup before restoration
      await this.validateBackup(backupPath);
      
      // Atomic replacement
      const tempPath = this.dbPath + '.temp';
      fs.copyFileSync(backupPath, tempPath);
      fs.renameSync(tempPath, this.dbPath);
      
      this.notifyRecoverySuccess();
    } catch (error) {
      this.notifyRecoveryFailure(error.message);
      throw error;
    }
  }
}
```

```typescript
class FileSystemPermissionManager {
  async checkPermissions(path: string): Promise<PermissionStatus> {
    try {
      await fs.access(path, fs.constants.R_OK | fs.constants.W_OK);
      return PermissionStatus.GRANTED;
    } catch (error) {
      switch (error.code) {
        case 'EACCES': // Permission denied
          return PermissionStatus.DENIED;
        case 'EPERM': // Operation not permitted
          return PermissionStatus.RESTRICTED;
        case 'ENOENT': // Path doesn't exist
          return PermissionStatus.PATH_NOT_FOUND;
        default:
          return PermissionStatus.UNKNOWN_ERROR;
      }
    }
  }

  async handlePermissionFailure(originalPath: string): Promise<string> {
    // Fallback hierarchy
    const fallbackPaths = [
      path.join(os.tmpdir(), 'sticky-notes'),
      path.join(os.homedir(), '.sticky-notes-fallback'),
      path.join(__dirname, 'emergency-storage')
    ];

    for (const fallbackPath of fallbackPaths) {
      try {
        await fs.mkdir(fallbackPath, { recursive: true });
        const status = await this.checkPermissions(fallbackPath);
        
        if (status === PermissionStatus.GRANTED) {
          this.notifyFallbackLocation(fallbackPath);
          return fallbackPath;
        }
      } catch (error) {
        console.warn(`Fallback path failed: ${fallbackPath}`, error);
      }
    }

    throw new Error('No writable storage location available');
  }
}
```

```typescript
class RuntimePermissionWatcher {
  private permissionWatcher: fs.FSWatcher | null = null;

  startMonitoring(storagePath: string): void {
    try {
      this.permissionWatcher = fs.watch(path.dirname(storagePath), (eventType, filename) => {
        if (filename && path.resolve(filename) === storagePath) {
          this.validatePermissions();
        }
      });
    } catch (error) {
      console.warn('Permission monitoring unavailable:', error.message);
    }
  }

  private async validatePermissions(): Promise<void> {
    const status = await this.permissionManager.checkPermissions(this.storagePath);
    
    if (status !== PermissionStatus.GRANTED) {
      // Pause auto-save and notify user
      this.pauseAutoSave();
      this.showPermissionError(status);
    }
  }
}
```

```typescript
    } catch (error) {
      // If we can't save crash context, at least log it
      console.error('Failed to save crash context:', error);
    }
  }
```

```typescript
        content,

      }));
      
      // Keep only last 10 temp files per note

    } catch (error) {
      console.warn('Failed to save temporary data:', error);
    }
  }

  async recoverUnsavedData(): Promise<RecoveryData[]> {
    const recoveryData: RecoveryData[] = [];
    
    try {
      const tempFiles = await fs.readdir(this.tempDir);
      
      for (const file of tempFiles) {
        if (file.startsWith('temp_')) {
          try {
            const filePath = path.join(this.tempDir, file);
            const data = JSON.parse(await fs.readFile(filePath, 'utf8'));
            
            // Verify data integrity
            if (this.calculateChecksum(data.content) === data.checksum) {
              recoveryData.push(data);
            }
          } catch (error) {
            console.warn(`Failed to recover temp file ${file}:`, error);
          }
        }
      }
      
      // Sort by timestamp, newest first
      recoveryData.sort((a, b) => 
        new Date(b.timestamp).getTime() - new Date(a.timestamp).getTime()
      );
      
    } catch (error) {
      console.error('Recovery data scan failed:', error);
    }
    
    return recoveryData;
  }
}
```

```typescript
class DiskSpaceMonitor {
  private spaceCheckInterval: NodeJS.Timeout | null = null;

  startMonitoring(): void {
    this.spaceCheckInterval = setInterval(() => {
      this.checkDiskSpace();
    }, 30000); // Check every 30 seconds
  }

  private async checkDiskSpace(): Promise<void> {
    try {
      const stats = await fs.statvfs(this.storagePath);
      const freeBytes = stats.bavail * stats.frsize;
      const freeGB = freeBytes / (1024 * 1024 * 1024);
      
      if (freeGB < 0.1) { // Less than 100MB free
        this.handleLowDiskSpace(freeBytes);
      } else if (freeGB < 1.0) { // Less than 1GB free
        this.warnLowDiskSpace(freeBytes);
      }
    } catch (error) {
      console.warn('Disk space check failed:', error);
    }
  }

  private handleLowDiskSpace(freeBytes: number): void {
    // Disable auto-save to prevent corruption
    this.autoSaveManager.disable();
    
    this.showDiskSpaceError({
      freeSpace: this.formatBytes(freeBytes),
      recommendedAction: 'Free up disk space to continue auto-saving'
    });
  }
}
```

### Auto-Save Queue Management

```typescript
class AutoSaveQueueManager {
  private saveQueue = new Map<string, AutoSaveTask>();
  private processing = false;
  private maxConcurrentSaves = 3;

  async queueAutoSave(noteId: string, content: string): Promise<void> {
    // Debounce rapid edits
    const existingTask = this.saveQueue.get(noteId);
    if (existingTask) {
      existingTask.content = content;
      existingTask.timestamp = Date.now();
      return;
    }

    const task: AutoSaveTask = {
      noteId,
      content,
      timestamp: Date.now(),
      attempts: 0,
      maxAttempts: 3
    };

    this.saveQueue.set(noteId, task);
    
    if (!this.processing) {
      this.processQueue();
    }
  }

  private async processQueue(): Promise<void> {
    this.processing = true;
    const activeSaves = new Set<Promise<void>>();

    try {
      while (this.saveQueue.size > 0 || activeSaves.size > 0) {
        // Start new saves up to max concurrent limit
        while (activeSaves.size < this.maxConcurrentSaves && this.saveQueue.size > 0) {
          const [noteId, task] = Array.from(this.saveQueue.entries())[0];
          this.saveQueue.delete(noteId);

          const savePromise = this.performSave(task)
            .catch(error => this.handleSaveError(task, error))
            .finally(() => activeSaves.delete(savePromise));

          activeSaves.add(savePromise);
        }

        // Wait for at least one save to complete
        if (activeSaves.size > 0) {
          await Promise.race(activeSaves);
        }
      }
    } finally {
      this.processing = false;
    }
  }

  private async performSave(task: AutoSaveTask): Promise<void> {
    try {
      // Check disk space before save
      await this.diskSpaceMonitor.ensureSufficientSpace();
      
      // Atomic write with temporary file
      const tempFile = `${this.getNotePath(task.noteId)}.tmp`;
      await fs.writeFile(tempFile, task.content);
      await fs.rename(tempFile, this.getNotePath(task.noteId));
      
      // Update database
      await this.db.run(
        'UPDATE notes SET content = ?, updated_at = ? WHERE id = ?',
        [task.content, new Date().toISOString(), task.noteId]
      );

      this.notifyAutoSaveSuccess(task.noteId);
    } catch (error) {
      throw new AutoSaveError(`Save failed for note ${task.noteId}: ${error.message}`);
    }
  }

  private async handleSaveError(task: AutoSaveTask, error: Error): Promise<void> {
    task.attempts++;
    
    if (task.attempts < task.maxAttempts) {
      // Exponential backoff retry
      const delay = Math.pow(2, task.attempts) * 1000;
      setTimeout(() => {
        this.saveQueue.set(task.noteId, task);
      }, delay);
    } else {
      // Max attempts exceeded - save to emergency location
      await this.saveToEmergencyLocation(task);
      this.notifyAutoSaveFailure(task.noteId, error.message);
    }
  }
}
```

### File Lock Conflict Resolution

```typescript
class FileLockManager {
  private lockFiles = new Map<string, NodeJS.Timeout>();

  async acquireFileLock(filePath: string, timeout = 5000): Promise<boolean> {
    const lockPath = `${filePath}.lock`;
    const lockId = Math.random().toString(36);

    try {
      // Try to create lock file exclusively
      await fs.writeFile(lockPath, lockId, { flag: 'wx' });
      
      // Set lock timeout
      const timeoutId = setTimeout(async () => {
        await this.releaseLock(filePath, lockId);
      }, timeout);
      
      this.lockFiles.set(filePath, timeoutId);
      return true;
    } catch (error) {
      if (error.code === 'EEXIST') {
        // Lock file exists - check if it's stale
        return await this.handleExistingLock(filePath, lockPath);
      }
      throw error;
    }
  }

  private async handleExistingLock(filePath: string, lockPath: string): Promise<boolean> {
    try {
      const lockStats = await fs.stat(lockPath);
      const lockAge = Date.now() - lockStats.mtime.getTime();
      
      // If lock is older than 30 seconds, consider it stale
      if (lockAge > 30000) {
        await fs.unlink(lockPath);
        return await this.acquireFileLock(filePath);
      }
      
      return false; // Active lock, cannot acquire
    } catch (error) {
      // Lock file disappeared, try again
      return await this.acquireFileLock(filePath);
    }
  }

  async releaseLock(filePath: string, lockId: string): Promise<void> {
    const lockPath = `${filePath}.lock`;
    
    try {
      const currentLock = await fs.readFile(lockPath, 'utf8');
      if (currentLock === lockId) {
        await fs.unlink(lockPath);
      }
    } catch (error) {
      // Lock file already removed or doesn't exist
    }
    
    const timeoutId = this.lockFiles.get(filePath);
    if (timeoutId) {
      clearTimeout(timeoutId);
      this.lockFiles.delete(filePath);
    }
  }
}
```

### Rollback Mechanism

```typescript
class SaveRollbackManager {
  private rollbackStack = new Map<string, RollbackPoint[]>();

  async createRollbackPoint(noteId: string): Promise<string> {
    const rollbackId = `rb_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
    
    try {
      // Get current state
      const currentNote = await this.db.get(
        'SELECT * FROM notes WHERE id = ?', 
        [noteId]
      );
      
      if (!currentNote) return rollbackId;

      const rollbackPoint: RollbackPoint = {
        id: rollbackId,
        noteId,
        content: currentNote.content,
        updatedAt: currentNote.updated_at,
        timestamp: new Date().toISOString()
      };

      // Store rollback point
      const rollbacks = this.rollbackStack.get(noteId) || [];
      rollbacks.push(rollbackPoint);
      
      // Keep only last 5 rollback points
      if (rollbacks.length > 5) {
        rollbacks.shift();
      }
      
      this.rollbackStack.set(noteId, rollbacks);
      
      // Persist to disk for crash recovery
      await this.persistRollbackPoint(rollbackPoint);
      
    } catch (error) {
      console.error('Failed to create rollback point:', error);
    }
    
    return rollbackId;
  }

  async rollback(noteId: string, rollbackId?: string): Promise<boolean> {
    const rollbacks = this.rollbackStack.get(noteId);
    if (!rollbacks || rollbacks.length === 0) return false;

    try {
      let rollbackPoint: RollbackPoint;
      
      if (rollbackId) {
        rollbackPoint = rollbacks.find(rb => rb.id === rollbackId);
        if (!rollbackPoint) return false;
      } else {
        // Use most recent rollback point
        rollbackPoint = rollbacks[rollbacks.length - 1];
      }

      // Restore previous state
      await this.db.run(
        'UPDATE notes SET content = ?, updated_at = ? WHERE id = ?',
        [rollbackPoint.content, rollbackPoint.updatedAt, noteId]
      );

      // Remove used rollback points
      const index = rollbacks.indexOf(rollbackPoint);
      rollbacks.splice(index);
      
      this.notifyRollbackSuccess(noteId, rollbackPoint.timestamp);
      return true;
      
    } catch (error) {
      console.error('Rollback failed:', error);
      this.notifyRollbackFailure(noteId, error.message);
      return false;
    }
  }
}
```
