```javascript
    if ('connection' in navigator) {
      navigator.connection.addEventListener('change', () => {
        this.connectionQuality = this.assessConnectionQuality();
        this.notifyListeners('quality-change', this.connectionQuality);
      });
    }
  }

  assessConnectionQuality() {
    const connection = navigator.connection;
    if (!connection) return 'unknown';
    
    const effectiveType = connection.effectiveType;
    const downlink = connection.downlink;
    const rtt = connection.rtt;
    
    if (effectiveType === 'slow-2g' || downlink < 0.5 || rtt > 2000) {
      return 'poor';
    } else if (effectiveType === '2g' || downlink < 1.5 || rtt > 1000) {
      return 'fair';
    } else if (effectiveType === '3g' || downlink < 5 || rtt > 500) {
      return 'good';
    } else {
      return 'excellent';
    }
  }

  async startConnectivityValidation() {
    setInterval(async () => {
      if (this.isOnline) {
        const actuallyOnline = await this.validateConnectivity();
        if (!actuallyOnline && this.isOnline) {
          this.handleOffline();
        }
      }
    }, 30000); // Validate every 30 seconds
  }

  async validateConnectivity() {
    try {
      const response = await fetch('/api/health', {
        method: 'HEAD',
        cache: 'no-cache',
        timeout: 5000
      });
      return response.ok;
    } catch (error) {
      return false;
    }
  }
}
```

```javascript
// Sync queue manager for offline operations
class SyncQueueManager {
  constructor(storage) {
    this.storage = storage;
    this.queue = [];
    this.processingQueue = false;
    this.maxRetries = 5;
    this.retryDelays = [1000, 2000, 5000, 10000, 30000]; // Exponential backoff
  }

  // Queue operations during offline periods
  async queueOperation(operation) {
    const queueItem = {
      id: generateUUID(),
      operation,
      timestamp: Date.now(),
      retryCount: 0,
      status: 'pending'
    };
    
    this.queue.push(queueItem);
    await this.persistQueue();
    
    // Try to process immediately if online
    if (navigator.onLine) {
      this.processQueue();
    }
    
    return queueItem.id;
  }

  async processQueue() {
    if (this.processingQueue) return;
    
    this.processingQueue = true;
    
    try {
      const pendingItems = this.queue.filter(item => 
        item.status === 'pending' && item.retryCount < this.maxRetries
      );
      
      for (const item of pendingItems) {
        await this.processQueueItem(item);
      }
      
      // Clean up completed and failed items
      this.queue = this.queue.filter(item => 
        item.status === 'pending' && item.retryCount < this.maxRetries
      );
      
      await this.persistQueue();
    } finally {
      this.processingQueue = false;
    }
  }

  async processQueueItem(item) {
    try {
      await this.executeOperation(item.operation);
      item.status = 'completed';
      this.notifySuccess(item);
    } catch (error) {
      item.retryCount++;
      
      if (this.isNetworkError(error)) {
        // Schedule retry for network errors
        setTimeout(() => {
          if (navigator.onLine) {
            this.processQueue();
          }
        }, this.retryDelays[Math.min(item.retryCount - 1, this.retryDelays.length - 1)]);
      } else {
        // Mark as failed for non-network errors
        item.status = 'failed';
        this.notifyError(item, error);
      }
    }
  }

  isNetworkError(error) {
    const networkErrors = [
      'ENETDOWN', 'ENETUNREACH', 'EHOSTUNREACH', 
      'ECONNRESET', 'ECONNREFUSED', 'ETIMEDOUT'
    ];
    return networkErrors.some(code => error.code === code) ||
           error.message?.includes('Network request failed') ||
           error.message?.includes('fetch');
  }
}
```

```javascript
class NetworkErrorHandler {
  static ERROR_CODES = {
    // Connection errors
    ENETDOWN: 'Network is down',
    ENETUNREACH: 'Network is unreachable', 
    EHOSTUNREACH: 'Host is unreachable',
    ECONNREFUSED: 'Connection refused',
    ECONNRESET: 'Connection reset by peer',
    ETIMEDOUT: 'Connection timed out',
    
    // HTTP errors
    HTTP_408: 'Request timeout',
    HTTP_429: 'Too many requests',
    HTTP_500: 'Internal server error',
    HTTP_502: 'Bad gateway',
    HTTP_503: 'Service unavailable',
    HTTP_504: 'Gateway timeout'
  };

  static handleError(error) {
    const errorCode = error.code || `HTTP_${error.status}`;
    const errorMessage = this.ERROR_CODES[errorCode] || 'Unknown network error';
    
    return {
      code: errorCode,
      message: errorMessage,
      isRetryable: this.isRetryableError(errorCode),
      retryAfter: this.getRetryDelay(errorCode),
      userMessage: this.getUserFriendlyMessage(errorCode)
    };
  }

  static isRetryableError(code) {
    const retryableErrors = [
      'ETIMEDOUT', 'ECONNRESET', 'ENETUNREACH',
      'HTTP_408', 'HTTP_429', 'HTTP_500', 'HTTP_502', 'HTTP_503', 'HTTP_504'
    ];
    return retryableErrors.includes(code);
  }

  static getRetryDelay(code) {
    const delays = {
      'HTTP_429': 60000, // 1 minute for rate limiting
      'HTTP_503': 30000, // 30 seconds for service unavailable
      'ETIMEDOUT': 10000, // 10 seconds for timeouts
      'default': 5000
    };
    return delays[code] || delays.default;
  }
}
```

```javascript
// Advanced conflict resolution with three-way merge
class ConflictResolver {
  constructor() {
    this.mergeStrategies = {
      'last-writer-wins': this.lastWriterWins.bind(this),
      'three-way-merge': this.threeWayMerge.bind(this),
      'user-choice': this.promptUserChoice.bind(this),
      'preserve-both': this.preserveBoth.bind(this)
    };
  }

  async resolveConflict(conflict) {
    const { localVersion, remoteVersion, baseVersion, strategy } = conflict;
    
    // Detect conflict type
    const conflictType = this.detectConflictType(localVersion, remoteVersion, baseVersion);
    
    switch (conflictType) {
      case 'no-conflict':
        return this.autoMerge(localVersion, remoteVersion, baseVersion);
      
      case 'simple-conflict':
        return await this.mergeStrategies[strategy](conflict);
      
      case 'complex-conflict':
        return await this.handleComplexConflict(conflict);
      
      default:
        throw new Error(`Unknown conflict type: ${conflictType}`);
    }
  }

  detectConflictType(local, remote, base) {
    // No changes
    if (this.contentEqual(local, base) && this.contentEqual(remote, base)) {
      return 'no-conflict';
    }
    
    // Only local changes
    if (this.contentEqual(remote, base) && !this.contentEqual(local, base)) {
      return 'no-conflict';
    }
    
    // Only remote changes
    if (this.contentEqual(local, base) && !this.contentEqual(remote, base)) {
      return 'no-conflict';
    }
    
    // Both changed - check for conflicts
    const localDiff = this.computeDiff(base, local);
    const remoteDiff = this.computeDiff(base, remote);
    
    if (this.hasOverlappingChanges(localDiff, remoteDiff)) {
      return 'complex-conflict';
    }
    
    return 'simple-conflict';
  }

  async threeWayMerge(conflict) {
    const { localVersion, remoteVersion, baseVersion } = conflict;
    
    try {
      // Compute diffs from base to both versions
      const localChanges = this.computeStructuredDiff(baseVersion, localVersion);
      const remoteChanges = this.computeStructuredDiff(baseVersion, remoteVersion);
      
      // Apply non-conflicting changes
      let mergedContent = { ...baseVersion };
      
      // Apply local changes
      for (const change of localChanges) {
        if (!this.conflictsWith(change, remoteChanges)) {
          mergedContent = this.applyChange(mergedContent, change);
        }
      }
      
      // Apply remote changes
      for (const change of remoteChanges) {
        if (!this.conflictsWith(change, localChanges)) {
          mergedContent = this.applyChange(mergedContent, change);
        }
      }
      
      // Handle remaining conflicts
      const remainingConflicts = this.findRemainingConflicts(localChanges, remoteChanges);
      
      if (remainingConflicts.length > 0) {
        return await this.handleRemainingConflicts(mergedContent, remainingConflicts);
      }
      
      return {
        success: true,
        mergedVersion: mergedContent,
        conflicts: []
      };
      
    } catch (error) {
      return {
        success: false,
        error: error.message,
        fallbackStrategy: 'user-choice'
      };
    }
  }

  computeStructuredDiff(base, current) {
    const changes = [];
    
    // Content changes
    if (base.content !== current.content) {
      changes.push({
        type: 'content',
        path: 'content',
        oldValue: base.content,
        newValue: current.content,
        operation: 'modify'
      });
    }
    
    // Metadata changes
    Object.keys(current.metadata || {}).forEach(key => {
      const baseValue = base.metadata?.[key];
      const currentValue = current.metadata[key];
      
      if (baseValue !== currentValue) {
        changes.push({
          type: 'metadata',
          path: `metadata.${key}`,
          oldValue: baseValue,
          newValue: currentValue,
          operation: baseValue === undefined ? 'add' : 'modify'
        });
      }
    });
    
    return changes;
  }
}
```

```javascript
// Advanced retry and circuit breaker implementation
class RetryManager {
  constructor() {
    this.circuits = new Map();
    this.defaultConfig = {
      maxRetries: 5,
      baseDelay: 1000,
      maxDelay: 30000,
      backoffMultiplier: 2,
      jitter: true,
      
      // Circuit breaker config
      failureThreshold: 5,
      recoveryTimeout: 60000,
      monitoringPeriod: 120000
    };
  }

  async executeWithRetry(operation, config = {}) {
    const finalConfig = { ...this.defaultConfig, ...config };
    const circuitKey = config.circuitKey || 'default';
    
    // Check circuit breaker
    if (this.isCircuitOpen(circuitKey)) {
      throw new Error('Circuit breaker is open - service unavailable');
    }
    
    let lastError;
    
    for (let attempt = 0; attempt <= finalConfig.maxRetries; attempt++) {
      try {
        const result = await this.executeWithTimeout(operation, finalConfig);
        
        // Success - reset circuit breaker
        this.recordSuccess(circuitKey);
        return result;
        
      } catch (error) {
        lastError = error;
        this.recordFailure(circuitKey, error);
        
        // Don't retry on non-retryable errors
        if (!this.isRetryableError(error)) {
          throw error;
        }
        
        // Don't wait after final attempt
        if (attempt < finalConfig.maxRetries) {
          const delay = this.calculateDelay(attempt, finalConfig);
          await this.sleep(delay);
        }
      }
    }
    
    throw lastError;
  }

  calculateDelay(attempt, config) {
    let delay = config.baseDelay * Math.pow(config.backoffMultiplier, attempt);
    delay = Math.min(delay, config.maxDelay);
    
    // Add jitter to prevent thundering herd
    if (config.jitter) {
      delay = delay * (0.5 + Math.random() * 0.5);
    }
    
    return delay;
  }

  async executeWithTimeout(operation, config) {
    const timeouts = {
      connection: config.connectionTimeout || 10000,
      read: config.readTimeout || 30000,
      total: config.totalTimeout || 60000
    };
    
    return new Promise((resolve, reject) => {
      const totalTimer = setTimeout(() => {
        reject(new Error(`Operation timed out after ${timeouts.total}ms`));
      }, timeouts.total);
      
      operation(timeouts)
        .then(result => {
          clearTimeout(totalTimer);
          resolve(result);
        })
        .catch(error => {
          clearTimeout(totalTimer);
          reject(error);
        });
    });
  }

  // Circuit breaker implementation
  isCircuitOpen(circuitKey) {
    const circuit = this.circuits.get(circuitKey);
    if (!circuit) return false;
    
    return circuit.state === 'open' && 
           Date.now() < circuit.nextAttempt;
  }

  recordFailure(circuitKey, error) {
    let circuit = this.circuits.get(circuitKey) || this.createCircuit(circuitKey);
    
    circuit.failures++;
    circuit.lastFailure = Date.now();
    
    // Check if we should open the circuit
    if (circuit.failures >= this.defaultConfig.failureThreshold) {
      circuit.state = 'open';
      circuit.nextAttempt = Date.now() + this.defaultConfig.recoveryTimeout;
      
      this.notifyCircuitOpen(circuitKey);
    }
    
    this.circuits.set(circuitKey, circuit);
  }

  recordSuccess(circuitKey) {
    let circuit = this.circuits.get(circuitKey);
    if (circuit) {
      circuit.failures = 0;
      circuit.state = 'closed';
      circuit.lastSuccess = Date.now();
      
      this.circuits.set(circuitKey, circuit);
    }
  }
}
```

```javascript
// Intelligent partial sync recovery
class PartialSyncRecovery {
  constructor(syncManager, storage) {
    this.syncManager = syncManager;
    this.storage = storage;
    this.recoveryCheckpoints = new Map();
  }

  async initiateRecovery() {
    try {
      // Get last successful sync timestamp
      const lastSyncTimestamp = await this.storage.getLastSyncTimestamp();
      
      // Identify items that failed to sync
      const failedItems = await this.identifyFailedSyncItems(lastSyncTimestamp);
      
      // Batch items by priority
      const batches = this.createRecoveryBatches(failedItems);
      
      // Process batches with checkpoints
      for (const batch of batches) {
        await this.processBatch(batch);
        await this.createCheckpoint(batch);
      }
      
      return {
        success: true,
        recoveredItems: failedItems.length,
        timestamp: Date.now()
      };
      
    } catch (error) {
      return {
        success: false,
        error: error.message,
        partialRecovery: await this.getRecoveryStatus()
      };
    }
  }

  async identifyFailedSyncItems(lastSyncTimestamp) {
    const localItems = await this.storage.getModifiedSince(lastSyncTimestamp);
    const failedItems = [];
    
    for (const item of localItems) {
      const syncStatus = await this.storage.getSyncStatus(item.id);
      
      if (syncStatus.status === 'failed' || 
          syncStatus.status === 'pending' ||
          syncStatus.lastAttempt < item.modifiedAt) {
        failedItems.push({
          ...item,
          syncStatus,
          priority: this.calculatePriority(item)
        });
      }
    }
    
    return failedItems.sort((a, b) => b.priority - a.priority);
  }

  createRecoveryBatches(items) {
    const batches = [];
    const batchSize = 10; // Configurable
    
    // Group high priority items in smaller batches
    const highPriority = items.filter(item => item.priority > 8);
    const normalPriority = items.filter(item => item.priority <= 8);
    
    // Process high priority items first in smaller batches
    for (let i = 0; i < highPriority.length; i += Math.ceil(batchSize / 2)) {
      batches.push({
        items: highPriority.slice(i, i + Math.ceil(batchSize / 2)),
        priority: 'high',
        timeout: 15000
      });
    }
    
    // Process normal priority in regular batches
    for (let i = 0; i < normalPriority.length; i += batchSize) {
      batches.push({
        items: normalPriority.slice(i, i + batchSize),
        priority: 'normal', 
        timeout: 30000
      });
    }
    
    return batches;
  }

  async processBatch(batch) {
    const promises = batch.items.map(item => 
      this.syncManager.syncItem(item).catch(error => ({
        item,
        error,
        success: false
      }))
    );
    
    const results = await Promise.allSettled(promises);
    
    // Update sync status for each item
    for (let i = 0; i < results.length; i++) {
      const result = results[i];
      const item = batch.items[i];
      
      if (result.status === 'fulfilled' && !result.value.error) {
        await this.storage.updateSyncStatus(item.id, 'synced');
      } else {
        await this.storage.updateSyncStatus(item.id, 'failed', {
          error: result.reason || result.value?.error,
          lastAttempt: Date.now()
        });
      }
    }
  }
}
```

```javascript
// Comprehensive token management system
class TokenManager {
  constructor(storage, authService) {
    this.storage = storage;
    this.authService = authService;
    this.refreshPromise = null;
    this.refreshThreshold = 300000; // 5 minutes before expiry
  }

  async getValidToken() {
    const token = await this.storage.getAccessToken();
    
    if (!token) {
      throw new AuthError('No access token available', 'NO_TOKEN');
    }
    
    // Check if token needs refresh
    if (this.shouldRefreshToken(token)) {
      return await this.refreshToken();
    }
    
    return token;
  }

  shouldRefreshToken(token) {
    try {
      const payload = this.decodeTokenPayload(token);
      const expiresAt = payload.exp * 1000; // Convert to milliseconds
      const now = Date.now();
      
      return (expiresAt - now) < this.refreshThreshold;
    } catch (error) {
      // Invalid token format
      return true;
    }
  }

  async refreshToken() {
    // Prevent multiple concurrent refresh attempts
    if (this.refreshPromise) {
      return await this.refreshPromise;
    }
    
    this.refreshPromise = this.executeTokenRefresh();
    
    try {
      const newToken = await this.refreshPromise;
      return newToken;
    } finally {
      this.refreshPromise = null;
    }
  }

  async executeTokenRefresh() {
    try {
      const refreshToken = await this.storage.getRefreshToken();
      
      if (!refreshToken) {
        throw new AuthError('No refresh token available', 'NO_REFRESH_TOKEN');
      }
      
      const response = await fetch('/api/auth/refresh', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json'
        },
        body: JSON.stringify({ refreshToken })
      });
      
      if (!response.ok) {
        if (response.status === 401 || response.status === 403) {
          // Refresh token is invalid - need full re-authentication
          await this.clearTokens();
          throw new AuthError('Refresh token expired', 'REFRESH_EXPIRED');
        }
        throw new Error(`Token refresh failed: ${response.status}`);
      }
      
      const tokenData = await response.json();
      
      // Store new tokens
      await this.storage.setAccessToken(tokenData.accessToken);
      if (tokenData.refreshToken) {
        await this.storage.setRefreshToken(tokenData.refreshToken);
      }
      
      return tokenData.accessToken;
      
    } catch (error) {
      // Clear tokens on refresh failure
      await this.clearTokens();
      throw error;
    }
  }

  async validateTokenBeforeSync(operation) {
    try {
      const token = await this.getValidToken();
      
      // Add authorization header to operation
      if (!operation.headers) {
        operation.headers = {};
      }
      operation.headers.Authorization = `Bearer ${token}`;
      
      return operation;
      
    } catch (error) {
      if (error.code === 'REFRESH_EXPIRED') {
        // Trigger re-authentication flow
        await this.triggerReAuthentication();
      }
      throw error;
    }
  }

  async triggerReAuthentication() {
    // Notify UI to show login prompt
    this.notifyAuthRequired();
    
    // Pause all sync operations
    this.pauseSync();
    
    // Wait for user to complete authentication
    return new Promise((resolve, reject) => {
      this.authService.onAuthComplete = (success, tokens) => {
        if (success) {
          this.resumeSync();
          resolve(tokens);
        } else {
          reject(new AuthError('Authentication failed', 'AUTH_FAILED'));
        }
      };
    });
  }
}
```

```javascript
// Bandwidth-aware sync optimization
class AdaptiveSyncManager {
  constructor() {
    this.connectionQuality = 'good';
    this.syncStrategies = {
      'poor': {
        maxConcurrentSyncs: 1,
        batchSize: 5,
        compressionLevel: 9,
        priorityOnly: true,
        syncInterval: 300000 // 5 minutes
      },
      'fair': {
        maxConcurrentSyncs: 2,
        batchSize: 10,
        compressionLevel: 6,
        priorityOnly: false,
        syncInterval: 120000 // 2 minutes
      },
      'good': {
        maxConcurrentSyncs: 3,
        batchSize: 20,
        compressionLevel: 3,
        priorityOnly: false,
        syncInterval: 60000 // 1 minute
      },
      'excellent': {
        maxConcurrentSyncs: 5,
        batchSize: 50,
        compressionLevel: 1,
        priorityOnly: false,
        syncInterval: 30000 // 30 seconds
      }
    };
  }

  async executeSyncWithAdaptation(items) {
    const strategy = this.syncStrategies[this.connectionQuality];
    
    // Filter items by priority if needed
    let syncItems = strategy.priorityOnly 
      ? items.filter(item => item.priority > 7)
      : items;
    
    // Apply compression
    syncItems = await this.compressPayloads(syncItems, strategy.compressionLevel);
    
    // Create batches
    const batches = this.createAdaptiveBatches(syncItems, strategy.batchSize);
    
    // Execute with concurrency limit
    const results = await this.executeConcurrentBatches(batches, strategy.maxConcurrentSyncs);
    
    return results;
  }

  async compressPayloads(items, compressionLevel) {
    if (compressionLevel === 1) return items; // No compression
    
    return items.map(item => ({
      ...item,
      content: this.compressContent(item.content, compressionLevel),
      compressed: true
    }));
  }

  compressContent(content, level) {
    // Simple compression simulation - in reality use gzip/brotli
    if (level >= 6) {
      // High compression - remove extra whitespace, minify
      return content
        .replace(/\s+/g, ' ')
        .replace(/\n\s*/g, '\n')
        .trim();
    } else if (level >= 3) {
      // Medium compression - remove extra line breaks
      return content.replace(/\n\s*\n/g, '\n');
    }
    
    return content; // Low/no compression
  }

  // Delta sync for large content
  async generateDeltaSync(oldContent, newContent) {
    const delta = this.computeTextDelta(oldContent, newContent);
    
    // Only use delta if it's significantly smaller
    if (delta.length < newContent.length * 0.7) {
      return {
        type: 'delta',
        delta: delta,
        size: delta.length
      };
    } else {
      return {
        type: 'full',
        content: newContent,
        size: newContent.length
      };
    }
  }

  // User controls for sync preferences
  setupUserControls() {
    return {
      syncFrequency: {
        realtime: 15000,
        frequent: 60000,
        moderate: 300000,
        conservative: 900000
      },
      
      dataUsage: {
        unlimited: { maxBatchSize: 100, compression: 1 },
        high: { maxBatchSize: 50, compression: 3 },
        moderate: { maxBatchSize: 20, compression: 6 },
        low: { maxBatchSize: 5, compression: 9 }
      },
      
      backgroundSync: {
        enabled: true,
        wifiOnly: false,
        priority: 'normal' // high, normal, low
      }
    };
  }

  async adaptToNetworkChange(newQuality) {
    this.connectionQuality = newQuality;
    
    // Cancel ongoing operations if network degraded significantly
    if (newQuality === 'poor' && this.ongoingSyncs.length > 1) {
      await this.cancelLowPrioritySyncs();
    }
    
    // Update sync interval
    this.updateSyncInterval();
    
    // Notify user of adaptation
    this.notifyNetworkAdaptation(newQuality);
  }
}
```
